居于绝对定位的瀑布流效果

发表于 2017-02-17 | 分类于 html/css/js |

导语

瀑布流效果见得很多了，项目中也经常用到，之前也写过，也用过插件，哈哈，但是不论怎样，基本实现方式可以分为三种，第一种是css3的多列布局，这种比较不专业，原因是排序是从左到后排列
比较少的运用在实际的项目中，第二种就是浮动布局，第三种就是绝对定位的布局，适用于屏幕自适应
这里用绝对定位来实现一番

这里数据使用的是gank.io的开源api

使用强大的jq来进行dom的操作会减少很多兼容性的问题

这里随便用对象的写法，也比较简单明了

var waterfall = {
                    more : true, //开关
                    page :1,  //开始页
                    getData: function(count,page){
                        var type = "福利";
                        var This = this;
                        $('.get_more').show();
                        $.ajax({
                            type:"get",
                            url:"http://gank.io/api/search/query/listview/category/"+type+"/count/"+count+"/page/"+page+"",
                            async:true,
                            success:function(data){                            
                                var result = data.results;
                                if(result.length>0){
                                    var li_box = "";
                                    for(var i=0; i<result.length; i++){
                                        li_box += "<div class='box_w'><div class=\"box\"><img src="+result[i].url+"><p>"+result[i].who+"</p></div></div>";
                                    }
                                    $('.waterfall_box').append(li_box);
                                    //设置图片的位置
                                    This.setImgPosition();
                                    //窗口变动时
                                       This.windowResize();
                                    more = true;
                                    $('.get_more').hide();
                                }else{
                                    $('.get_more').html('没有更多了...');
                                }
                            }
                        });    
                    },
                    setImgPosition:function(){
                        var This= this;
                        //获取所有的li
                        var allLi = $('.waterfall_box .box_w');
                        var con_w= $('.waterfall_box').width();
                        //能排多少张
                        var num = Math.floor(con_w/allLi.outerWidth()); 

                        var elesHeight = []; //每个元素的高度
                        var colsHeight = [];  //每一列的高度，起始高度都是为0

                        $.each(allLi, function(index,item) {
                            eleHeight = allLi.eq(index).outerHeight();
                            elesHeight.push(eleHeight);
                        });

                       for(var j=0;j<num;j++){
                            colsHeight.push(0);
                       }
                       //设置每一张图片
                       $('.waterfall_box .box_w').each(function(index,item){
                             var colShould = This.getMinHeightCol(colsHeight);  //高度最小的
                             var tempTop = colsHeight[colShould];
                             var tempLeft = colShould*allLi.outerWidth();
                             $(this).css({
                                 'left':tempLeft,
                                 'top':tempTop
                             });
                             colsHeight[colShould]+=$(this).outerHeight();  //将当前的高度附加到之前的该列的高度中
                       });

                    },
                    windowResize:function (){
                        var This = this;
                         $(window).on('scroll',function(){
                             var scrollTop = $(this).scrollTop();
                             var docHeight = $(document).height();

                             if(docHeight-$(window).height()-scrollTop<100&&more==true){
                                 This.page ++;
                                 more = false;
                                 This.getData(30,This.page);
                          }
                       });
                       $(window).on('resize',function(){
                            This.setImgPosition();  //重新设置位置
                       })
                    },
                    getMinHeightCol:function(arr){
                         var minHeight = Math.min.apply(null,arr);
                            for(var i in arr){
                                if(arr[i]== minHeight){
                                    return i;
                                }
                            }
                            return 0;
                    }

                }
