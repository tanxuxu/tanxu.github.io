一个demo解析angular的directive常应用

发表于 2016-10-28 | 分类于 vue/angular/ionic |

导语

angular的指令应该是angular中最难的一个内容，很多人开发中干脆尽量少用指令，这个demo列出常见的angular的指令的用法，包括控制器之间的传输，指令与
指令之间的数据传输，element的使用。

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>Document</title>
    <style>
        .error-class {
            background: #f00;
            width: 200px;
            height: 200px;
            color: #fff;
        }
    </style>
    <script src=" http://cdn.bootcss.com/angular.js/1.6.3/angular.min.js "></script>
    <script>
        angular.module('app', [])
            .controller('indexCtrl', function($scope) {
                $scope.sayName = function() {
                    alert(222);
                }
            })
            .directive('hello', function() {
                return {
                    restrict: 'E',
                    replace: true,
                    template: '<div>我是一个div</div>',
                    link: function(scope, elements, attrs) {
                        elements.bind('click', function() {
                            scope.$apply(attrs.name);
                        })
                    }
                }
            })
            .directive('food', function() {
                return {
                    scope: {},
                    restrict: 'E',
                    controller: function($scope) {
                        $scope.foods = [];
                        this.addApple = function() {
                            $scope.foods.push('apple');
                        }
                        this.addOrange = function() {
                            $scope.foods.push('orange');
                        }
                    },
                    link: function(scope, elements, attrs) {
                        elements.bind('click', function() {
                            console.log(scope.foods);
                        })
                    }
                }
            })
            .directive('apple', function() {
                return {
                    require: 'food',
                    link: function(scope, elements, attrs, foodCtrl) {
                        elements.bind('click', function() {
                            foodCtrl.addApple();
                        })
                    }
                }
            })

        .directive('hello1', function() {
            return {
                restrict: 'E',
                replace: true,
                template: '<div><input ng-model="text"/><div>{{text}}</div></div>',
                link: function(scope, elements, attrs) {
                    scope.$watch('text', function(newVal) {
                        if (newVal === 'error') {
                            elements.css({
                                'background': '#f00'
                            })
                        }
                    })
                }
            }
        })
    </script>

</head>

<body ng-app="app">
    <div ng-controller="indexCtrl">
        <hello name='sayName();'></hello>

        <food apple>asdfs</food>
        <hello1>测试hello</hello1>
    </div>
</body>

</html>
